Chart.js는 자바스크립트로 차트를 쉽게 그릴 수 있게 해주는 라이브러리입니다.
HTML의 <canvas> 태그를 이용해서 막대그래프, 선그래프, 원그래프 등을 만들 수 있습니다.

<!DOCTYPE html>
<html lang="ko">
<head>
    <meta charset="UTF-8">
    <title>Chart.js 기초</title>
</head>
<body>

    <!-- 차트를 그릴 캔버스 -->
    <canvas id="myChart"></canvas>

    <!-- Chart.js CDN -->
    <script src="https://cdn.jsdelivr.net/npm/chart.js"></script> -- chart.js 라이브러리 호출

    <script>
        // 캔버스 객체 선택
        const ctx = document.getElementById('myChart'); -- html에는 <canvas id="myChart">를 가져옴

        // 차트 생성
        new Chart(ctx, { --  new Chart 차트를 만들 객체 생성, ctx : 캠퍼스에 차트 그리기
            type: 'bar',  -- 차트 종류('bar' : 막대그래프, 'line' : 선그래프, 'pie' : 원그래프, 'doughunt' : 도너그래프)
            data: { -- 차트 위에 그려질 실제 데이터
                labels: ['1월', '2월', '3월', '4월'], -- x축이나 항목 이름
                datasets: [{ -- 실제 값들
                    label: '매출', -- 범례에 표시
                    data: [10, 20, 30, 25], -- 실제 그래프 값
                    borderWidth: 1 -- 막대 테두리 두께
                }]
            },
            options: { --그래프 디자인
                scales: { -- x축 / y축 설정
                    y: {
                        beginAtZero: true -- y축 시작 값을 0으로 설정 / false : 데이터 최소값부터 시작
                    }
                }
            }
        });
    </script>

</body>
</html>

추가 설명

차트 크기 조정(캠퍼스 크기 지정)
width : 가로 크기
height : 세로 크기

<canvas id="myChart" width="600" height="300"></canvas>

<canvas id="myChart1" width="600" height="300"></canvas>
css를 이용한 크기 조정 방법

<canvas id="myChart1"></canvas>

<style>
#myChart1 {
    width: 600px;
    height: 300px;
}
</style>

반응형 만들기

options: {
    responsive: true 화면 크기에 따라 자동 조정됨 / false : canvas 크기 그대로 유지 됨
}

차트 제목 삽입

options: {
    plugins: {
        title: {
            display: true,           // 제목 표시 여부
            text: '월별 매출 그래프' // 제목 내용
            color : 'blue' 
            size : 20
        }
    }
}

축제목 삽입
scales: {
    y: {
        title: {
            display: true,
            text: '매출액'
        }
    }
}

scales: {
    x: {
        title: {
            display: true,
            text: '월'
        }
    }
}

<!DOCTYPE html>
<html lang="ko">
<head>
    <meta charset="UTF-8">
    <title>Chart.js 전체 예제</title>

    <style>
        /* 차트 크기 설정 */
        #myChart {
            width: 600px;
            height: 300px;
        }
    </style>
</head>
<body>

    <!-- 차트 영역 -->
    <canvas id="myChart"></canvas>

    <!-- Chart.js -->
    <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

    <script>
        const ctx = document.getElementById('myChart');

        new Chart(ctx, {
            type: 'bar',

            data: {
                labels: ['1월', '2월', '3월', '4월'],
                datasets: [{
                    label: '매출',
                    data: [10, 20, 30, 25],
                    borderWidth: 1
                }]
            },

            options: {
                responsive: false, // 고정 크기

                plugins: {
                    title: {
                        display: true,
                        text: '월별 매출 그래프',
                        font: {
                            size: 20
                        }
                    }
                },

                scales: {
                    x: {
                        title: {
                            display: true,
                            text: '월'
                        }
                    },
                    y: {
                        beginAtZero: true,
                        title: {
                            display: true,
                            text: '매출액'
                        }
                    }
                }
            }
        });
    </script>

</body>
</html>

플라스크에서 보낸 자료를 이용하여 차트를 작성 방법


플라스크에서 보내기
names = ["홍길동", "김철수", "이영희"]
avg_list = [90, 80, 70]
overall_avg = 80

return render_template(
    "chart.html",
    names=names,
    avg_list=avg_list,
    overall_avg=overall_avg
)

HTML에서 받기
<script>
    const names = {{ names|tojson }};  -- 파이썬 데이터를 자바스크립트가 이해할 수 있는 JSON 형태로 바꿔주는 역활
    const avgList = {{ avg_list|tojson }};
    const overallAvg = {{ overall_avg|tojson }};



    new Chart(ctx, {
        data: {
            labels: names,
            datasets: [{
                label: "학생 평균",
                data: avgList
            }]
        }
    });
</script>
